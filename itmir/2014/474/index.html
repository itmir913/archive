아래 메소드로 HTML 태그 삭제가 가능합니다.

public String RemoveHTMLTag(String changeStr){
    if(changeStr != null && !changeStr.equals("")){
        changeStr = changeStr.replaceAll("<(/)?([a-zA-Z]*)(\\s[a-zA-Z]*=[^>]*)?(\\s)*(/)?>", "");
    }else{
        changeStr = "";
    }
    return changeStr;
}
